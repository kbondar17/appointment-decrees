	function subQuerySearch (query, e) {
		if (null == e)
			return false;
		var code = e.keyCode ? e.keyCode : window.event.keyCode; 
		if (13 == code) {
			if(findValue== '' || null == findValue){
				findValue = document.getElementById('find_str').value;
				var result = window.findParentText(window, true);
				if (null != result) {
					setNoTextFoundSignal();
					return false;
				} else {
					document.getElementById('find_from').value = 0;
					clearNoTextFoundSignal();
					return true;
				}
			} else {

				return false;
			}
		} else {
			findValue = '';
			document.getElementById('find_from').value = 1;
		}
		return false;
	}

	function find(subQuery, findDirect) {
		var fromParam;
		var toParam;
		if(findDirect) {
			toParam = 0;
			fromParam = 1;
		} else {
			toParam = 1;
			fromParam = 2;
		}
		document.getElementById('find_to').selectedIndex = toParam;
		if (subQuery == null) 
			subQuery = false;
		var res = window.findParentText(window, subQuery);
		if (null != res) {
			if (fromParam == document.getElementById('find_from').value) {
				setNoTextFoundSignal();
				return false;
			}
			document.getElementById('find_from').value = fromParam;
			res = window.findParentText(window, subQuery);
			if (null != res) {
				setNoTextFoundSignal();
				return false;
			} else if (subQuery) {
				document.getElementById('find_from').value = 0;
				clearNoTextFoundSignal();
				return true;
			}
		}
		document.getElementById('find_from').value = 0;
		clearNoTextFoundSignal();
		return false;
	}

var selected = -1;

function goToContext(napr)
{	
	var ifr = document.getElementById('list');
	var doc = ifr.contentDocument || ifr.contentWindow.document;
	var spans = doc.getElementsByTagName("SPAN");
	if (spans == null)
		return;
	
	var i = (selected != -1 ? selected : 0) + napr;
	var new_selected = -1;
	while (i >= 0 && i < spans.length) {
		if ((spans.item(i).className + "").indexOf('bookmark') != -1) {
			new_selected = i;
			break;
		}
		i += napr;
	}
	if (i < 0) {
		if (!confirm("Это первое вхождение. Перейти на последнее?")) {
			return;
		}
		i = spans.length - 1;
	} else if (i >= spans.length) {
		if ( !confirm("Это последнее вхождение. Перейти на первое?") ) {
			return;
		}
		i = 0;
	}
	if (new_selected == -1) {
		while (i >= 0 && i < spans.length) {
			if ((spans.item(i).className + "").indexOf('bookmark') != -1) {
				new_selected = i;
				break;
			}
			i += napr;
		}
	}
	if (new_selected != -1) {
		if (selected != -1) spans.item(selected).className = 'bookmark';
		spans.item(new_selected).className = 'searched_cur';
		selected = new_selected;
		spans.item(new_selected).scrollIntoView(false);
	}
	
}
